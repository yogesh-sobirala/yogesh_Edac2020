4)  Write a Javascript program to sort elements of an array using sorting algorithm

var arr = [7,4,77,5,1,9,3,72];
function sor_arr(arr)
{
    var temp=0;
    var i,j;
    for(i=0;i<arr.length-1;i++)
    {
        for(j=i+1;j<arr.length;j++)
        {
            if(arr[i]>arr[j])
            {
                temp=arr[i];
                arr[i]=arr[j];
                arr[j]=temp;
            }
        }
    }
    return arr;
}
